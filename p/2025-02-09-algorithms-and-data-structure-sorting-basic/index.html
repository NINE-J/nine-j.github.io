function bubbleSort(A) {
    let n = A.length;
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (A[j] > A[j + 1]) {
                // A[j]ê³¼ A[j + 1]ì˜ ìœ„ì¹˜ë¥¼ ë³€ê²½í•œë‹¤.
                let temp = A[j];
                A[j] = A[j + 1];
                A[j + 1] = temp;
            }
        }
    }
    return A;
}

function selectionSort(A) {
    let n = A.length;
    for (let i = 0; i < n - 1; i++) {
        let minIndex = i;
        for (let j = i + 1; j < n; j++) {
            if (A[j] < A[minIndex]) {
                minIndex = j;
            }
        }
        if (minIndex != i) {
            // A[i]ê³¼ A[minIndex]ì˜ ìœ„ì¹˜ë¥¼ ë³€ê²½í•œë‹¤.
            let temp = A[i];
            A[i] = A[minIndex];
            A[minIndex] = temp;
        }
    }
    return A;
}

function insertionSort(A) {
    let n = A.length;
    for (let i = 1; i < n; i++) {
        let key = A[i];
        let j = i - 1;
        while (j >= 0 && A[j] > key) {
            A[j + 1] = A[j];
            j = j - 1;
        }
        A[j + 1] = key;
    }
    return A;
}
